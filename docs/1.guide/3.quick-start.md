# Quick Start

> Detect agents, discover skills, and parse sources in a few lines.

## Detect a Running Agent

```ts
import { detectCurrentAgent, detectInstalledAgents } from 'unagent/env'

const agent = detectCurrentAgent()
if (agent) {
  console.log(`Running in ${agent.config.name}`)
}

const installed = detectInstalledAgents()
console.log(`Found ${installed.length} agents`)
```

## Discover Skills

```ts
import { discoverSkills, parseSkillMd } from 'unagent/skill'

const skills = discoverSkills('~/.claude/skills', { recursive: true })

const parsed = parseSkillMd(`---
name: My Skill
globs: ["*.ts"]
---
Instructions for the agent...
`)
```

## Install Skills

```ts
import { installSkill } from 'unagent/skill'

await installSkill({ source: 'unjs/unagent' })
await installSkill({ source: './local/skills', mode: 'symlink' })
```
