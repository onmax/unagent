## Usage

```ts [detect.ts]
import { agents, detectInstalledAgents, getAgentConfig } from 'unagent/env'

// Get all agent IDs
const ids = Object.keys(agents) // ['claude-code', 'cursor', ...]

// Get specific agent config
const claude = getAgentConfig('claude-code')
// → { name: 'Claude Code', configDir: '~/.claude', rulesFile: 'CLAUDE.md', ... }

// Detect which agents are installed
const installed = detectInstalledAgents()
for (const agent of installed) {
  console.log(`${agent.config.name}: detected via ${agent.detected}`)
}
```

## Agent Config Type

```ts
interface AgentConfig {
  name: string // Display name
  configDir: string // Config directory (supports ~)
  rulesFile?: string // Rules/instructions file
  skillsDir?: string // Skills directory name
  legacySkillsDirs?: string[] // Legacy skills directory names
  envDetect?: string[] // Env vars for detection
  description?: string // Short description
}
```
