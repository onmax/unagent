# Nitro

> Attach a vector client to the Nitro context.

## Setup

```ts
import { vectorNitroPlugin } from 'unagent/vector/nitro'

export default defineNitroPlugin((nitroApp) => {
  vectorNitroPlugin({
    contextKey: 'vector',
    defaultProvider: 'upstash',
    providers: {
      upstash: {
        name: 'upstash',
        url: process.env.UPSTASH_VECTOR_URL!,
        token: process.env.UPSTASH_VECTOR_TOKEN!,
        namespace: 'chunks',
      },
    },
  })(nitroApp)
})
```

## Usage

```ts
import { getVector } from 'unagent/vector/nitro'

export default defineEventHandler(async (event) => {
  const vector = await getVector(event)
  const results = await vector.search('hello')
  return { results }
})
```
