# Browserbase

> Hosted browser sessions through Browserbase + Playwright CDP.

## Setup

```bash
npm i @browserbasehq/sdk playwright
```

## Usage

```ts
import { createBrowser } from 'unagent/browser'

const browser = await createBrowser({
  provider: {
    name: 'browserbase',
    apiKey: process.env.BROWSERBASE_API_KEY,
    projectId: process.env.BROWSERBASE_PROJECT_ID,
    session: {
      keepAlive: true,
    },
  },
})

const session = await browser.newSession()
const page = await session.newPage({ url: 'https://example.com' })
const html = await page.extract({ kind: 'html' })

await session.close() // Requests Browserbase session release
await browser.close()
```
